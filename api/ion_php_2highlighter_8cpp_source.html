00101 {
00102     foreach (const HighlightingRule &rule, highlightingRules) {
00103         QRegExp expression(rule.pattern);
00104         int index = expression.indexIn(text);
00105         while (index >= 0) {
00106             int length = expression.matchedLength();
00107             setFormat(index, length, rule.format);
00108             index = expression.indexIn(text, index + length);
00109         }
00110     }
00111     setCurrentBlockState(0);
00112 
00113     int startIndex = 0;
00114     if (previousBlockState() != 1)
00115         startIndex = commentStartExpression.indexIn(text);
00116 
00117     while (startIndex >= 0) {
00118         int endIndex = commentEndExpression.indexIn(text, startIndex);
00119         int commentLength;
00120         if (endIndex == -1) {
00121             setCurrentBlockState(1);
00122             commentLength = text.length() - startIndex;
00123         } else {
00124             commentLength = endIndex - startIndex
00125                     + commentEndExpression.matchedLength();
00126         }
00127         setFormat(startIndex, commentLength, multiLineCommentFormat);
00128         startIndex = commentStartExpression.indexIn(text, startIndex + commentLength);
00129     }
